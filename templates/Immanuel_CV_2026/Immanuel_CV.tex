\section*{Checklist}
\addcontentsline{toc}{section}{Checklist}
% ==========================================================

Use this checklist before submitting to ensure all documents are included.

\begin{itemize}[leftmargin=2em, label={$\square$}]
  \item \textbf{Personal Data:} Passport or identity card
  \item \textbf{Personal Data:} Name change evidence (if applicable)
  \item \textbf{Education:} Diploma(s) for all declared degrees
  \item \textbf{Education:} Transcript(s) for all declared degrees
  \item \textbf{Education:} Transcript for ongoing studies (if applicable)
  \item \textbf{Education:} Professional certifications (if declared)
  \item \textbf{Home Experience:} Proof for each declared experience (signed, dated, with employer header and exact period)
  \item \textbf{International Experience:} Proof for each declared experience (signed, dated, with employer/institution header and exact period)
  \item \textbf{Languages:} Certificate for each language declared at B1 or above
\end{itemize}
